class BaseMetric:
    """Base class for computing evaluation metrics.

    Attributes:
        name (str): The name of the metric.
        task (str): The type of task, either 'classification' or 'regression'.
        models (Dict[str, object]): A dictionary of initialized models for the task.
    """



[docs]
    def __init__(self, name: str, task: str) -> None:
        """Initializes the BaseMetric.

        Args:
            name (str): Metric name.
            task (str): Task type ('classification' or 'regression').

        Raises:
            ValueError: If task is not 'classification' or 'regression'.
        """
        if task not in {"classification", "regression"}:
            raise ValueError("Task must be 'classification' or 'regression'.")

        self.name = name
        self.task = task
        self.models = self._initialize_models()

    def _initialize_models(
        self,
    ) -> Dict[
        str,
        Union[
            RandomForestClassifier,
            LogisticRegression,
            GradientBoostingClassifier,
            RandomForestRegressor,
            LinearRegression,
            GradientBoostingRegressor,
        ],
    ]:
        """Initializes task-specific models.

        Returns:
            Dict[str, object]: Dictionary mapping model names to model instances.
        """
        return {
            "classification": {
                "Random Forest": RandomForestClassifier(),
                "Logistic Regression": LogisticRegression(max_iter=1000),
                "Gradient Boosting": GradientBoostingClassifier(),
            },
            "regression": {
                "Random Forest": RandomForestRegressor(),
                "Linear Regression": LinearRegression(),
                "Gradient Boosting": GradientBoostingRegressor(),
            },
        }[self.task]

    def train_and_predict(
        self,
        X_train: np.ndarray,
        y_train: np.ndarray,
        X_test: np.ndarray,
        y_test: np.ndarray,
    ) -> Dict[str, Dict[str, Union[np.ndarray, None]]]:
        """Trains models and generates predictions.

        Args:
            X_train (np.ndarray): Training feature data.
            y_train (np.ndarray): Training labels.
            X_test (np.ndarray): Test feature data.
            y_test (np.ndarray): Test labels.

        Returns:
            Dict[str, Dict[str, Union[np.ndarray, None]]]:
                A dictionary containing predictions and probabilities.
        """
        results = {}

        for model_name, model in self.models.items():
            model.fit(X_train, y_train)
            predictions = model.predict(X_test)
            probabilities = (
                model.predict_proba(X_test) if self.task == "classification" else None
            )
            results[model_name] = {
                "predictions": predictions,
                "probabilities": probabilities,
            }

        return results

    def compute(
        self,
        X_train: np.ndarray,
        y_train: np.ndarray,
        X_test: np.ndarray,
        y_test: np.ndarray,
    ) -> float:
        """Computes the regression metric.

        Args:
            X_train (np.ndarray): Training feature data.
            y_train (np.ndarray): Training labels.
            X_test (np.ndarray): Test feature data.
            y_test (np.ndarray): Test labels.

        Returns:
            float: Computed metric value.
        """
        results = self.train_and_predict(X_train, y_train, X_test, y_test)
        return np.mean(
            [self._metric_func(y_test, res["predictions"]) for res in results.values()]
        )

    def compute(
        self,
        X_train: np.ndarray,
        y_train: np.ndarray,
        X_test: np.ndarray,
        y_test: np.ndarray,
    ) -> float:
        """Computes the classification metric.

        Args:
            X_train (np.ndarray): Training feature data.
            y_train (np.ndarray): Training labels.
            X_test (np.ndarray): Test feature data.
            y_test (np.ndarray): Test labels.

        Returns:
            float: Computed metric value.
        """
        results = self.train_and_predict(X_train, y_train, X_test, y_test)
        return np.mean(
            [
                self._metric_func(y_test, res["predictions"], res.get("probabilities"))
                for res in results.values()
            ]
        )
